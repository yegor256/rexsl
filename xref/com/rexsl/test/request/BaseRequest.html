57  /**
58   * Base implementation of {@link Request}.
59   *
60   * @author Yegor Bugayenko (yegor@tpc2.com)
61   * @version $Id$
62   * @since 0.8
63   * @checkstyle ClassDataAbstractionCoupling (500 lines)
64   * @see Request
65   * @see Response

67  @Immutable
68  @EqualsAndHashCode(of = { "home", "mtd", "hdrs", "content" })
69  @Loggable(Loggable.DEBUG)
70  @SuppressWarnings("PMD.TooManyMethods")
71  final class BaseRequest implements Request {
72  
73      /**
74       * Wire to use.
75       */
76      private final transient Wire wire;
77  
78      /**
79       * Request URI.
80       */
81      private final transient String home;
82  
83      /**
84       * Method to use.
85       */
86      private final transient String mtd;
87  
88      /**
89       * Headers.
90       */
91      private final transient Array<Map.Entry<String, String>> hdrs;
92  
93      /**
94       * Body to use.
95       */
96      private final transient byte[] content;

98      /**
99       * Public ctor.
100      * @param wre Wire
101      * @param uri The resource to work with
102      */
103     BaseRequest(final Wire wre, final String uri) {
104         this(
105             wre, uri,
106             new Array<Map.Entry<String, String>>(),
107             Request.GET, ArrayUtils.EMPTY_BYTE_ARRAY
108         );
109     }

111     /**
112      * Public ctor.
113      * @param wre Wire
114      * @param uri The resource to work with
115      * @param headers Headers
116      * @param method HTTP method
117      * @param body HTTP request body
118      * @checkstyle ParameterNumber (5 lines)
119      */
120     BaseRequest(final Wire wre, final String uri,
121         final Iterable<Map.Entry<String, String>> headers,
122         final String method, final byte[] body) {
123         this.wire = wre;
124         URI addr = URI.create(uri);
125         if (addr.getPath().isEmpty()) {
126             addr = UriBuilder.fromUri(addr).path("/").build();
127         }
128         this.home = addr.toString();
129         this.hdrs = new Array<Map.Entry<String, String>>(headers);
130         this.mtd = method;
131         this.content = ArrayUtils.clone(body);
132     }
133 
134     @Override
135     @NotNull
136     public RequestURI uri() {
137         return new BaseRequest.BaseURI(this, this.home);
138     }

140     @Override
141     public Request header(
142         @NotNull(message = "header name can't be NULL") final String name,
143         @NotNull(message = "header value can't be NULL") final Object value) {
144         return new BaseRequest(
145             this.wire,
146             this.home,
147             this.hdrs.with(new ImmutableHeader(name, value.toString())),
148             this.mtd,
149             this.content
150         );
151     }

153     @Override
154     public Request reset(
155         @NotNull(message = "header name can't be NULL") final String name) {
156         final Collection<Map.Entry<String, String>> headers =
157             new LinkedList<Map.Entry<String, String>>();
158         final String key = ImmutableHeader.normalize(name);
159         for (final Map.Entry<String, String> header : this.hdrs) {
160             if (!header.getKey().equals(key)) {
161                 headers.add(header);
162             }
163         }
164         return new BaseRequest(
165             this.wire,
166             this.home,
167             headers,
168             this.mtd,
169             this.content
170         );
171     }
172 
173     @Override
174     public RequestBody body() {
175         return new BaseRequest.BaseBody(this, this.content);
176     }

178     @Override
179     public Request method(
180         @NotNull(message = "method can't be NULL") final String method) {
181         return new BaseRequest(
182             this.wire,
183             this.home,
184             this.hdrs,
185             method,
186             this.content
187         );
188     }
189 
190     @Override
191     public Response fetch() throws IOException {
192         final long start = System.currentTimeMillis();
193         final Response response = this.wire.send(
194             this, this.home, this.mtd,
195             this.hdrs, this.content
196         );
197         Logger.info(
198             this,
199             "#fetch(%s %s): completed in %[ms]s [%d %s]: %s",
200             this.mtd,
201             URI.create(this.home).getPath(),
202             System.currentTimeMillis() - start,
203             response.status(),
204             response.reason(),
205             this.home
206         );
207         return response;
208     }

210     @Override
211     public <T extends Wire> Request through(final Class<T> type) {
212         try {
213             return new BaseRequest(
214                 type.getDeclaredConstructor(Wire.class).newInstance(
215                     this.wire
216                 ),
217                 this.home,
218                 this.hdrs,
219                 this.mtd,
220                 this.content
221             );
222         } catch (InstantiationException ex) {
223             throw new IllegalStateException(ex);
224         } catch (IllegalAccessException ex) {
225             throw new IllegalStateException(ex);
226         } catch (InvocationTargetException ex) {
227             throw new IllegalStateException(ex);
228         } catch (NoSuchMethodException ex) {
229             throw new IllegalStateException(ex);
230         }
231     }

233     @Override
234     public String toString() {
235         final URI uri = URI.create(this.home);
236         final StringBuilder text = new StringBuilder("HTTP/1.1 ")
237             .append(this.mtd).append(' ')
238             .append(uri.getPath())
239             .append(" (")
240             .append(uri.getHost())
241             .append(")\n");
242         for (final Map.Entry<String, String> header : this.hdrs) {
243             text.append(
244                 Logger.format(
245                     "%s: %s\n",
246                     header.getKey(),
247                     header.getValue()
248                 )
249             );
250         }
251         return text.append('\n')
252             .append(RequestBody.Printable.toString(this.content))
253             .toString();
254     }

256     /**
257      * Base URI.
258      */
259     @Immutable
260     @EqualsAndHashCode(of = "address")
261     @Loggable(Loggable.DEBUG)
262     private static final class BaseURI implements RequestURI {
263         /**
264          * URI encapsulated.
265          */
266         private final transient String address;
267         /**
268          * Base request encapsulated.
269          */
270         private final transient BaseRequest owner;
271         /**
272          * Public ctor.
273          * @param req Request
274          * @param uri The URI to start with
275          */
276         public BaseURI(final BaseRequest req, final String uri) {
277             this.owner = req;
278             this.address = uri;
279         }
280         @Override
281         public String toString() {
282             return this.address;
283         }
284         @Override
285         public Request back() {
286             return new BaseRequest(
287                 this.owner.wire,
288                 this.address,
289                 this.owner.hdrs,
290                 this.owner.mtd,
291                 this.owner.content
292             );
293         }
294         @Override
295         public URI get() {
296             return URI.create(this.owner.home);
297         }
298         @Override
299         public RequestURI set(@NotNull(message = "URI can't be NULL")
300             final URI uri) {
301             return new BaseRequest.BaseURI(this.owner, uri.toString());
302         }
303         @Override
304         public RequestURI queryParam(
305             @NotNull(message = "param name can't be NULL") final String name,
306             @NotNull(message = "value can't be NULL") final Object value) {
307             return new BaseRequest.BaseURI(
308                 this.owner,
309                 UriBuilder.fromUri(this.address)
310                     .queryParam(name, "{value}")
311                     .build(value).toString()
312             );
313         }
314         @Override
315         public RequestURI queryParams(@NotNull(message = "map can't be NULL")
316             final Map<String, String> map) {
317             final UriBuilder uri = UriBuilder.fromUri(this.address);
318             final Object[] values = new Object[map.size()];
319             int idx = 0;
320             for (final Map.Entry<String, String> pair : map.entrySet()) {
321                 uri.queryParam(pair.getKey(), String.format("{x%d}", idx));
322                 values[idx] = pair.getValue();
323                 ++idx;
324             }
325             return new BaseRequest.BaseURI(
326                 this.owner,
327                 uri.build(values).toString()
328             );
329         }
330         @Override
331         public RequestURI path(
332             @NotNull(message = "path can't be NULL") final String segment) {
333             return new BaseRequest.BaseURI(
334                 this.owner,
335                 UriBuilder.fromUri(this.address)
336                     .path(segment)
337                     .build().toString()
338             );
339         }
340         @Override
341         public RequestURI userInfo(
342             @NotNull(message = "info can't be NULL") final String info) {
343             return new BaseRequest.BaseURI(
344                 this.owner,
345                 UriBuilder.fromUri(this.address)
346                     .userInfo(info)
347                     .build().toString()
348             );
349         }
350     }

355     @Immutable
356     @EqualsAndHashCode(of = "text")
357     @Loggable(Loggable.DEBUG)
358     private static final class BaseBody implements RequestBody {
359         /**
360          * Content encapsulated.
361          */
362         private final transient byte[] text;
363         /**
364          * Base request encapsulated.
365          */
366         private final transient BaseRequest owner;
367         /**
368          * Public ctor.
369          * @param req Request
370          * @param body Text to encapsulate
371          */
372         public BaseBody(final BaseRequest req, final byte[] body) {
373             this.owner = req;
374             this.text = ArrayUtils.clone(body);
375         }
376         @Override
377         public String toString() {
378             return RequestBody.Printable.toString(this.text);
379         }
380         @Override
381         public Request back() {
382             return new BaseRequest(
383                 this.owner.wire,
384                 this.owner.home,
385                 this.owner.hdrs,
386                 this.owner.mtd,
387                 this.text
388             );
389         }
390         @Override
391         public String get() {
392             return new String(this.text, Charsets.UTF_8);
393         }
394         @Override
395         public RequestBody set(@NotNull(message = "content can't be NULL")
396             final String txt) {
397             return this.set(txt.getBytes(Charsets.UTF_8));
398         }
399         @Override
400         public RequestBody set(@NotNull(message = "body can't be NULL")
401             final byte[] txt) {
402             return new BaseRequest.BaseBody(this.owner, txt);
403         }
404         @Override
405         public RequestBody formParam(
406             @NotNull(message = "name can't be NULL") final String name,
407             @NotNull(message = "value can't be NULL") final Object value) {
408             try {
409                 return new BaseRequest.BaseBody(
410                     this.owner,
411                     new StringBuilder(this.get())
412                         .append(name)
413                         .append('=')
414                         .append(
415                             URLEncoder.encode(
416                                 value.toString(), CharEncoding.UTF_8
417                             )
418                         )
419                         .append('&')
420                         .toString()
421                         .getBytes(Charsets.UTF_8)
422                 );
423             } catch (UnsupportedEncodingException ex) {
424                 throw new IllegalStateException(ex);
425             }
426         }
427         @Override
428         public RequestBody formParams(
429             @NotNull(message = "map of params can't be NULL")
430             final Map<String, String> params) {
431             RequestBody body = this;
432             for (final Map.Entry<String, String> param : params.entrySet()) {
433                 body = body.formParam(param.getKey(), param.getValue());
434             }
435             return body;
436         }
437     }
