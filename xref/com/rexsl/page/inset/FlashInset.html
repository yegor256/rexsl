53  /**
54   * Flash message (through cookie).
55   *
56   * @author Yegor Bugayenko (yegor@tpc2.com)
57   * @version $Id$
58   * @since 0.4.8
59   * @see BasePage
60   * @see <a href="http://www.rexsl.com/rexsl-page/inset-flash.html">Flash messages in RESTful interfaces</a>
61   */
62  @ToString
63  @EqualsAndHashCode(of = "resource")
64  @Loggable(Loggable.DEBUG)
65  @SuppressWarnings("PMD.TooManyMethods")
66  public final class FlashInset implements Inset {

68      /**
69       * ImmutableHeader name.
70       */
71      private static final String HEADER = "X-Rexsl-Flash";
72  
73      /**
74       * Cookie name.
75       */
76      private static final String COOKIE = "Rexsl-Flash";

78      /**
79       * The resource.
80       */
81      private final transient Resource resource;

83      /**
84       * Public ctor.
85       * @param res The resource
86       */
87      public FlashInset(@NotNull final Resource res) {
88          this.resource = res;
89      }
90  
91      @Override
92      public void render(@NotNull final BasePage<?, ?> page,
93          @NotNull final Response.ResponseBuilder builder) {
94          if (this.resource.httpHeaders().getCookies()
95              .containsKey(FlashInset.COOKIE)) {
96              final FlashInset.Flash cookie = new FlashInset.Flash(
97                  this.resource.httpHeaders().getCookies().get(FlashInset.COOKIE)
98              );
99              page.append(
100                 FlashInset.bundle(
101                     cookie.level(),
102                     cookie.message(),
103                     cookie.msec()
104                 )
105             );
106             builder.cookie(
107                 new CookieBuilder(this.resource.uriInfo().getBaseUri())
108                     .name(FlashInset.COOKIE)
109                     .build()
110             );
111         }
112     }

117      * @param msg Message
118      * @param msec Time spent
119      * @return JaxbBundle injectable into the page

121     public static JaxbBundle bundle(@NotNull final Level level,
122         @NotNull final String msg, final long msec) {
123         return new JaxbBundle("flash")
124             .add("message", msg).up()
125             .add("level", level.toString()).up()
126             .add("msec", Long.toString(msec)).up();
127     }

140     /**
141      * Create an exception that will redirect to the page with an error message.
142      *
143      * <p>The difference between this method and a static
144      * {@link #forward(URI,String,Level)} is in its awareness of a resource,
145      * which is forwarding. Key benefit is that a non-static method adds extra
146      * value to the cookie, which is time consumed by the resource until the
147      * redirect happened.
148      *
149      * @param uri The URI to forward to
150      * @param message The message to show as error
151      * @param level Message level
152      * @return The exception to throw
153      */
154     public WebApplicationException redirect(@NotNull final URI uri,
155         @NotNull final String message, @NotNull final Level level) {
156         return new FlashException(
157             Response.status(HttpURLConnection.HTTP_SEE_OTHER)
158                 .location(uri)
159                 .cookie(
160                     new FlashInset.Flash(
161                         uri, message, level,
162                         System.currentTimeMillis() - this.resource.started())
163                 )
164                 .header(FlashInset.HEADER, message)
165                 .entity(message)
166                 .build(),
167             message,
168             level
169         );
170     }

222     /**
223      * The cookie.
224      */
225     public static final class Flash extends NewCookie {
226         /**
227          * Total elements expected in a cookie text.
228          */
229         private static final int TOTAL = 3;
230         /**
231          * The message.
232          */
233         private final transient String msg;
234         /**
235          * Level of message.
236          */
237         private final transient Level lvl;
238         /**
239          * Milliseconds consumed (or -1).
240          */
241         private final transient long millis;
242         /**
243          * Public ctor, from a cookie encoded text.
244          * @param cookie The cookie
245          */
246         public Flash(@NotNull final Cookie cookie) {
247             super(FlashInset.COOKIE, cookie.getValue());
248             Validate.notBlank(cookie.getValue(), "cookie value is blank");
249             final String decoded = FlashInset.Flash.decode(this.getValue());
250             final String[] parts = decoded.split(":", FlashInset.Flash.TOTAL);
251             Validate.isTrue(
252                 parts.length == FlashInset.Flash.TOTAL,
253                 // @checkstyle LineLength (1 line)
254                 "can't decode cookie '%s' (decoded='%s'), %d parts expected but %d found",
255                 this.getValue(),
256                 decoded,
257                 FlashInset.Flash.TOTAL,
258                 parts.length
259             );
260             Validate.isTrue(
261                 parts[0].matches("INFO|WARNING|SEVERE"),
262                 "can't detect level of cookie '%s' (decoded='%s')",
263                 this.getValue(),
264                 decoded
265             );
266             this.lvl = Level.parse(parts[0]);
267             Validate.isTrue(
268                 parts[1].matches("-1|\\d+"),
269                 "can't parse milliseconds in cookie '%s' (decoded='%s')",
270                 this.getValue(),
271                 decoded
272             );
273             this.millis = Long.parseLong(parts[1]);
274             Validate.notBlank(
275                 parts[2],
276                 "empty message in cookie '%s' (decoded='%s')",
277                 this.getValue(),
278                 decoded
279             );
280             this.msg = parts[2];
281         }
282         /**
283          * Public ctor, from exact values.
284          * @param base Base URI where we're using it
285          * @param message The message
286          * @param level The level
287          */
288         public Flash(@NotNull final URI base,
289             @NotNull final String message, @NotNull final Level level) {
290             this(base, message, level, -1);
291         }
292         /**
293          * Public ctor, from exact values.
294          * @param base Base URI where we're using it
295          * @param message The message
296          * @param level The level
297          * @param msec Milliseconds consumed
298          * @checkstyle ParameterNumber (5 lines)
299          */
300         public Flash(@NotNull final URI base,
301             @NotNull final String message, @NotNull final Level level,
302             final long msec) {
303             super(
304                 new CookieBuilder(base)
305                     .name(FlashInset.COOKIE)
306                     .value(FlashInset.Flash.encode(message, level, msec))
307                     .temporary()
308                     .build()
309             );
310             Validate.notBlank(message, "flash message can't be empty");
311             this.msg = message;
312             this.lvl = level;
313             Validate.isTrue(
314                 msec >= 0L || msec == -1L,
315                 "milliseconds can either be positive or equal to -1"
316             );
317             this.millis = msec;
318         }
319         /**
320          * Get message.
321          * @return The message
322          */
323         public String message() {
324             return this.msg;
325         }
326         /**
327          * Get color of it.
328          * @return The color
329          */
330         public Level level() {
331             return this.lvl;
332         }
333         /**
334          * Milliseconds consumed.
335          * @return The amount of them
336          */
337         public long msec() {
338             return this.millis;
339         }
340         @Override
341         public String getPath() {
342             return "/";
343         }
344         /**
345          * Encode message and color.
346          * @param message The message
347          * @param level The level
348          * @param msec Milliseconds consumed
349          * @return Encoded cookie value
350          */
351         private static String encode(final String message, final Level level,
352             final long msec) {
353             try {
354                 return Base64.encodeBase64URLSafeString(
355                     String.format("%s:%d:%s", level, msec, message)
356                         .getBytes(CharEncoding.UTF_8)
357                 );
358             } catch (UnsupportedEncodingException ex) {
359                 throw new IllegalStateException(ex);
360             }
361         }
362         /**
363          * Decode text.
364          * @param text The text to decode (cookie value)
365          * @return Decoded value
366          */
367         private static String decode(final String text) {
368             try {
369                 return new String(
370                     Base64.decodeBase64(text),
371                     CharEncoding.UTF_8
372                 );
373             } catch (UnsupportedEncodingException ex) {
374                 throw new IllegalStateException(ex);
375             }
376         }
377 
378     }
